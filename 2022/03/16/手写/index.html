function co(generatorFunc) {
	return new Promise((resolve, reject) => {
		// ç”Ÿæˆå™¨
		const gen = generatorFunc();

		function step(fn) {
			let res = null;
			try {
				res = fn.next();
			} catch (e) {
				reject(e);
			}
			if (res.done) {
				return resolve(res.value);
			}
			Promise.resolve(res.value).then(
				(v) => {
					step(() => gen.next(v));
				},
				(r) => {
					step(() => gen.throw(r));
				}
			);
		}

		step(() => gen.next());
	});
}
