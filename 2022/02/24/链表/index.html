链表

本文最后更新于：2022年3月21日 下午

特性

  • 数组需要申请连续的内存空间，链表可以使用“零散的”内存空间
  • 插入/删除时间复杂度O(1)
  • 随机访问时间复杂度O(n)

结构

  • 零散的内存块称为结点
  • 记录下一个结点地址的指针叫做后继指针next
  • headNode
  • tailNode.next = null

种类

  • 单链表
  • 循环链表

tailNode.next = headNode

适合处理环状结构的数据

  • 双向链表

    结点拥有两个指针
    curNode.prev
    curNode.next

    O(1)时间复杂度即可访问前序结点

  • 双向循环链表

    headNode.prev = tailNode
    tailNode.next = headNode

// 单向链表
class Node {
  constructor(element) {
    this.element = element;
    this.next = null;
  }
}
class linkedList {
  constructor() {
    this.head = null;
    this.length = 0;
  }
  // 追加
  append(element) {
    const n = new Node(element);
    let cur = this.head;
    if (!this.head) {
      this.head = n;
    } else {
      while (cur.next) {
        cur = cur.next;
      }
      cur.next = n;
      this.length++;
    }
  }
  // 查找
  search(element) {
    let cur = this.head;
    if (!cur) return false;
    while (cur) {
      if (cur.element === element) {
        return true;
      }
      cur = cur.next;
    }
    return false;
  }
  // 插入
  // position之前插入新node
  insert(position, element) {
    if (position >= 0 && position <= this.length) {
      let n = new Node(element);
      if (position === 0) {
        n.next = this.head;
        this.head = n;
      } else {
        let cur = this.head,pre
        let index = 0;
        while (index < position) {
          pre = cur;
          cur = cur.next;
          index++;
        }
        pre.next = n;
        n.next = cur;
      }
      this.length++;
    } else {
      return false;
    }
  }
  // 删除
  remove(element) {
    let cur = this.head,
      pre = this.head
    if (!this.head) return false
    while (cur) {
      if (cur.element === element) {
        cur = cur.next;
        pre.next = cur;
        this.length -- 
      }
      pre = cur
      cur = cur.next;
    }
  }
}



// 双向链表
function DubLinkedList(params) {
  const Node = function (element) {
    this.prev = null;
    this.next = null;
    this.element = element;
  };
  this.head = null;
  this.tail = null;
  this.length = 0;

  this.append = (element) => {
    let newNode = new Node(element);
    let cur = this.head;
    if (!cur) {
      this.head = newNode;
      this.tail = newNode;
    } else {
      while (cur.next) {
        cur = cur.next;
      }
      cur.next = newNode
      newNode.prev = cur
      this.tail = newNode
    }
    this.length++;
  }

  this.insert = (position,element) => {
    if(position>=0 && position <=this.length){
      let newNode = new Node(element)
      let cur = this.head,pre
      let index = 0

      if(position===0){
        //头
        if(!cur){
          this.head = newNode;
          this.tail = newNode;
        }else{
          newNode.next = this.head
          cur.prev = newNode
          this.head = newNode
        }
      }else if(position===this.length){
        //尾巴
        cur = this.tail
        cur.prev = newNode
        newNode.next = cur
        this.tail = newNode
      }else{
        while(index<position){
          pre = cur
          cur = cur.next
          index++
        }
        cur.prev = newNode
        pre.next = newNode
        newNode.next = cur
        newNode.prev = pre
      }
      this.length ++
    }else{
      return false
    }
  }
  this.remove = position => {
    if(position>=0 && position<this.length && this.length>0){
      let cur = this.head,pre,index = 0
      if(position===0){
        if(this.length === 1){
          this.head = null
          this.tail = null
        }else{
          cur = cur.next
          cur.prev = null
          this.head = cur
        }
      }else if(position===this.length-1){
        cur = this.tail
        pre = cur.prev
        pre.next = null
        this.tail = pre
      }else{
        while(index<position){
          pre = cur
          cur = cur.next
          index ++
        }
        cur = cur.next
        pre.next = cur
        cur.prev = pre
      }
      this.length --
      return cur.element
    }else{
      return null
    }
  }
  this.search = element =>{
    if(!this.head) return false
    let cur = this.head
    while(cur){
      if(cur.element === element){
        return true
      }
      cur = cur.next
    }
    return false
  }
}

function isLoop2(head){
  if(!head || !head.next) return false
  let fast = head.next.next
  let slow = head.next
  while(slow!==fast){
    if(!fast || !fast.next){
      return false
    }
    fast = fast.next.next
    slow = slow.next
  }
  return true
}

  this.value = value
  this.next = null
}
function LRULinkedList(maxSize){
  this.maxSize = maxSize
  this.head = null
  this.curSize = 0

  this.visit = value => {
    const n = new Node(value)
    let cur = this.head,pre = this.head
    let preCatch,curCatch
    if(!cur){
      this.head = n
      this.curSize = 1
      return 
    }
    if(cur.value===value) return 
    while(cur){
      if(cur.value === value){
          // 找到了
          // 从原来位置删除并作为头
          pre.next = cur.next
          cur.next = this.head
          return
      }
      preCatch = pre
      curCatch = cur
      pre = cur
      cur = cur.next
    }
    // 没找到
    if(this.curSize<maxSize){
      // 插入头部
      n.next = this.head
      this.head = n
      this.curSize ++
    }else{
      // 删除尾部结点
      // 插入头部
      preCatch.next = null
      curCatch.next = this.head
      this.head = curCatch
    }
  }
}

function isPalindrome(head){
  if(!head || !head.next) return true
  const stack = []
  let slow = head
  let fast = head
  while(fast && fast.next && fast.next.next){
    slow = slow.next
    fast = fast.next.next
  }
  let mid  = slow
  while (mid.next) {
    stack.push(mid.next)
    mid = mid.next
  }
  while(stack.length>0){
    if(head.val !== stack.pop().val) return false
    head = head.next
  }
  return true
}

function mergeLinkedList(l1,l2){
  if(l1===null){
    return l2
  }
  if(l2===null){
    return l1
  }
  if(l1.val<=l2.val){
    l1.next = mergeLinkedList(l1.next,l2)
    return l1
  }else{
    l2.next = mergeLinkedList(l2.next,l1)
    return l2
  }
}

var mergeTwoLists = function(l1, l2) {
   const n = new ListNode(-1)
   let cur = n

           l2 = l2.next

       cur.next = l2

function commonPrefix2(headA,headB){
  if(!headA || !headB) return null
  // 连个链表同时移动指针
  let n1 = headA
  let n2 = headB
  while(n1 || n2){
    if(n1===n2) return n1
    //指针到头后跳到对面的开头
    n1 = n1 === null ? headB : n1.next
    n2 = n2 === null ? headA : n2.next
  }
  return null
}

function firstUniqChar(s){
  if(!s) return ' '
  const map = new Map()
  for(let [i,v] of Array.from(s).entries()){
    if(map.has(v)){
      map.set(v,-1)
    }else{
      map.set(v,i)
    }
  }
  for(let key of map.keys()){
    if(map.get(key)!==-1){
      return key
    }
  }
  return ' '
}
