function firstUniqChar(s){
  if(!s) return ' '
  const map = new Map()
  for(let [i,v] of Array.from(s).entries()){
    if(map.has(v)){
      map.set(v,-1)
    }else{
      map.set(v,i)
    }
  }
  for(let key of map.keys()){
    if(map.get(key)!==-1){
      return key
    }
  }
  return ' '
}
