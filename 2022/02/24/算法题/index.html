function myPow(x, n) {
	function _pow(base, ex) {
		if (ex === 0) return 1;
		if (ex === 1) return base;

		const splitPow = _pow(base, Math.floor(ex / 2));
		return ex % 2 ? base * splitPow * splitPow : splitPow * splitPow;
	}
	const res = _pow(x, Math.abs(n));
	return n < 0 ? 1 / res : res;
}
