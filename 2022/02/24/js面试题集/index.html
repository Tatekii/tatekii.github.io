const Event = {
	on: function (name, cb) {
		if (!this.handlers) {
			Object.defineProperty(this, "handlers", {
				value: {},
				enumerable: false,
				configurable: true,
				writable: true,
			});
		}
		if (!this.handlers[name]) {
			this.handlers[name] = [];
		}
		this.handles[name].push(cb);
	},
	emit: function (name, ...args) {
		if (this.handlers[name]) {
			this.handlers[name].forEach((cb) => {
				cb(...args);
			});
		}
	},
};
class Events {
	constructor() {
		this.events = new Map();
	}
	addEvent(key, fn, isOnce, ...args) {
		const curEvent = this.events.get(key) ? this.events.get(key) : this.events.set(key, new Map()).get(key);
		curEvent.set(fn, (...args2) => {
			fn(...args, ...args2);
			if (isOnce) this.off(key, fn);
		});
	}
	on(key, fn, ...args) {
		this.addEvent(key, fn, false, ...args);
	}
	once(key, fn, ...args) {
		this.addEvent(key, fn, true, ...args);

		this.events.get(key).delete(fn);

	}
}

// r是否在l原型链上游
function instance_of(l, r) {
	r = r.prototype;
	l = Object.getPrototypeOf(l); // l =  l.__proto__
	while (true) {
		if (l === null) return false;
		if (r === l) return true;
		l = Object.getPrototypeOf(l);
	}
}

// 防抖
// 连续触发函数只执行一次
// 连续触发过程中计时器被重置
// NOTE:连续点击按钮(登陆),窗口resize,文本框实时保存,搜索框输入过程中的下拉候选项
function debounce(func, delay, immediate = true, ...args1) {
	if (typeof func !== "function") throw new TypeError("must apply to an Function");

	let timer;
	function _debounce(...args2) {
		if (timer) {
			clearTimeout(timer);
		}

		if (immediate) {
			const callNow = !timer;
			timer = setTimeout(() => {
				timer = null;
			}, delay);
			if (callNow) func.apply(this, [...args1, ...args2]);
		} else {
			timer = setTimeout(() => {
				func.apply(this, [...args1, ...args2]);
				clearTimeout(timer);
				timer = null;
			}, delay);
		}
	}
	_debounce.cancel = () => {
		clearTimeout(timer);
		timer = null;
	};
	return _debounce;
}

// 节流
// 函数以间隔固定时间执行
// 连续触发过程中计时器不会被重置
// NOTE:scrollEvent,clickEvent,resizeEvent,记录多媒体播放进度
function throttle(func, delay, tail = true, ...args1) {
	//tail 使用定时器
	//tail = false 使用时间戳
	if (typeof func !== "function") throw new TypeError("must apply to an Function");
	let timer = null;
	let pre = 0;
	if (tail) {
		function _throttle(...args2) {
			if (timer) return;
			timer = setTimeout(() => {
				func.apply(this, [...args1, ...args2]);
				clearTimeout(timer);
				timer = null;
			}, delay);
		}
		_throttle.cancel = () => {
			clearTimeout(timer);
			timer = null;
		};
		return _throttle;
	} else {
		return function (...args2) {
			const now = +Date.now();
			if (now - pre > delay) {
				func.apply(this, [...args1, ...args2]);
				pre = now;
			}
		};
	}
}

function asyncPool(limit, arr, func) {
	// arr参数队列
	// func执行函数
	let i = 0;
	const len = arr.length;
	const ret = []; // 所有任务
	const executing = []; // 运行中

	const enqueue = function () {
		if (i > len - 1) return Promise.resolve();

		const item = arr[i++];
		const p = Promise.resolve().then(() => func(item));
		const e = p.then(() => executing.splice(executing.indexOf(e), 1)); // 自己resolve后从并发池中删除

		ret.push(p);
		executing.push(e);
		console.log(executing);

		let r = Promise.resolve();
		if (executing.length >= limit) {
			r = Promise.race(executing);
			//使用race，并发池中只要有一个promise返回resolve，则继续调用enqueue方法添加新的进去
		}

		return r.then(() => enqueue());
	};
	return enqueue().then(() => Promise.all(ret));
}

  • 实现重试功能
/*
 *@functionName: retry
 *@fn: 执行函数
 *@interval: 重试间隔
 *@limit: 重试次数
 */
function retry(fn, interval, limit) {
	return new Promise((resolve, reject) => {
		function attempt() {
			const p = Promise.resolve().then(() => fn());
			p.then(resolve).catch((err) => {
				if (limit-- <= 1) {
					reject("打到重试最大次数");
				} else {
					const timer = setTimeout(() => {
						attempt();
						clearTimeout(timer);
					}, interval);
				}
			});
		}
		attempt();
	});
}

Function.prototype._call = function (context, ...args) {
	context = context || window;
	const caller = Symbol("caller");
	context[caller] = this;
	let result = context[caller](...args);
	Reflect.deleteProperty(context, caller);
	return result;
};

Function.prototype._apply = function (context, args) {
	context = context || window;
	const caller = Symbol("caller");
	context[caller] = this;
	let result = context[caller](...args);
	Reflect.deleteProperty(context, caller);
	return result;
};

// 返回一个原函数的拷贝，并拥有指定的this值和初始参数。
Function.prototype._bind = function (context, ...args1) {
	context = context || window;
	const originFn = this;
	const boundFn = function (...args2) {
		return originFn.apply(boundFn.prototype.isPrototypeOf(this) ? this : context, [
			// 如果绑定函数被new调用
			// 则this指向生的成新对象
			...args1,
			...args2,
		]);
	};

	// if (this.prototype) {
	//   notBFn.prototype = this.prototype;
	// }
	// boundFn.prototype = new notBFn();
	boundFn.prototype = Object.create(originFn.prototype);
	// 使用notBFn间接连接boundFn和originFn原型链
	return boundFn;
};

function _setTimeout(cb, timeout, ...args) {
	const start = +new Date();
	function loop() {
		const timer = window.requestAnimationFrame(loop);
		const now = +new Date();
		if (now - start >= timeout) {
			cb.apply(this, args);
			window.cancelAnimationFrame(timer);
		}
	}
	window.requestAnimationFrame(loop);
}

// setInterval 的问题
// 定时往事件队列中添加回调函数，如果主线程运行时间过长到大量回调超时，则有可能同时执行大量异步回调，违背使用意图
function _setInterval(fn,timeout,...args){
  const controller = {
    stop = false
    // 控制器
  }
  function _interval(){
    if(!controller.stop){
      fn.apply(this,args)
      setTimeout(_interval,timeout)
    }
  }
  setTimeout(_interval,timeout)
  return controller
}

  • Ajax
const request = (method, url, data) => {
	return new Promise((resolve, reject) => {
		const xhr = new XMLHttpRequest();
		xhr.onreadystatechange = function () {
			if (this.readyState !== 4) return;
			if (this.status === 200) {
				resolve(JSON.parse(this.response));
			} else {
				reject(new Error(this.statusText));
			}
		};
		xhr.onerror = function () {
			reject(new Error("error"));
		};
		xhr.responseType = "json";
		xhr.setRequestHeader("Accept", "application/json");
		if (method === "GET") {
			xhr.open("GET", url + "?" + data, true);
			xhr.send();
		} else if (method === "POST") {
			xhr.setRequestHeader("Content-type", "application/x-www-form-urlencode");
			xhr.open("POST", url);
			xhr.send(encodeURIComponent(data));
		}
	});
};

  • 数组的全排列
function permute(arr) {
	const res = [];
	// 回溯算法
	function recursion(path, set) {
		if (path.length === arr.length) {
			return res.push([...path]); // 潜拷贝一次
		}
		// 使用path记录回溯的路径
		// [1,2,3,4]
		// path[1,2]，set[1,2]
		// 1,2,已经用过
		// 分别遍历3，4
		// 添加[1,2,3,4],[1,2,4,3]进结果
		for (let num of arr) {
			if (set.has(num)) continue;
			path.push(num);
			set.add(num);
			recursion(path, set);
			path.pop();
			set.delete(num);
		}
	}
	recursion([], new Set());
	return res;
}

function binaryFind(arr, target, s = 0, e = arr.length - 1) {
	const i = Math.floor((s + e) / 2);
	if (arr[i] === target) {
		return i;
	} else if (arr[i] > target) {
		return binaryFind(arr, target, s, i - 1);
	} else if (arr[i] < target) {
		return binaryFind(arr, target, i + 1, e);
	}
	return -1;
}

function binaryFind2(arr, target) {
	let s = 0,
		e = arr.length - 1,
		mid;
	while (s <= e) {
		mid = Math.floor((s + e) / 2);
		if (arr[mid] === target) {
			return mid;
		} else if (arr[mid] > target) {
			e = mid - 1;
		} else if (arr[mid] < target) {
			s = mid + 1;
		}
	}
	return -1;
}
  • 扁平化转换
// 实现一个 fn 函数
const o = { a: 1, b: [1, 2, { c: true }], d: { e: 2, f: 3 }, g: null };
// fn(o) => 扁平化转换
// { "a": 1, "b[0]": 1, "b[1]": 2, "b[2].c": true, "d.e": 2, ... }

function objectFlat(obj) {
	const res = {};
	function flat(obj, path) {
		if (typeof obj !== "object") {
			res[path] = obj;
		} else {
			for (let key of Object.keys(obj)) {
				const cur = obj[key];
				if (Array.isArray(cur)) {
					for (let i = 0; i < cur.length; i++) {
						flat(cur[i], path ? path + "." + key + "[" + i + "]" : key + "[" + i + "]");
					}
				} else if (typeof obj[key] === "object") {
					flat(cur, path ? path + "." + key : key);
				} else {
					res[path ? path + "." + key : key] = cur;
				}
			}
		}
	}
	flat(obj, "");
	return res;
}
