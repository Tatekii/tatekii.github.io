npm 包本地开发

  • 链接本地 npm 包的依赖库

    // xxx-lib dir
npm link
// current module package.json
dependencies:{
+    "xxx-lib":"^1.0.0"
}
// current module dir
npm link xx-lib

  • 解除本地 xxx-lib 和当前包的链接

    // xxx-lib dir
npm unlink
// current module dir
rm -rf node_modules
npm unlink xxx-lib

  • 链接当前项目到全局node_modules

    npm link/unlink

  • 相对于当前node_modules操作

    npm link B // 添加依赖B
npm unlink B //  移除依赖B

Lerna

基于 Git+Npm 的多 package 项目管理工具

  1. 解决原生脚手架开发中的重复操作

    • 多 package 本地 link
    • 多 package 依赖安装
    • 多 package 单元测试
    • 多 package 代码提交
    • 多 package 代码发布

  2. 解决版本一致性

    • package 版本一致
    • 相互依赖版本升级

lerna 开发 module 流程

  1. 项目创建

    npm init
npm i lerna
lerna init

  2. 创建 package

    lerna create // 创建package
lerna add //安装依赖
lerna //链接依赖

  3. 开发测试

    lerna exec // 执行shell脚本
lerna run
lerna clean // 清空依赖
lerna bootstrap // 重装依赖

  4. 发布上线

    lerna version // 提升版本号
lerna change //  查看变更
lerna diff // 查看diff
lerna public // 发布

Plop

小型的脚手架工具,=>

使用

  1. 项目中安装 plop

    yarn add plop

  2. plopfile.js

    module.exports = plop => {
  plop.setGenerator('component',{
    description:'create a component',
    prompts:[
      {
        type:'input',
        name:'name',
        message:'your component name',
        default:'My Component'
      }
    ],
    actions:[ // 多个action
      {
        type:'add'// 添加文件
        path:'src/components/{{name}}/{{name}}.js',
        templateFile:'plop-templates/component.hbs'

      {
        type:'add',
        path:'src/components/{{name}}/{{name}}.css',
        templateFile:'plop-template/component.css.hbs'
      }
    ]
  })
}

异步 task

grunt task 默认是同步代码

module.exports = (grunt) => {
	// 异步任务要获取this,不能用=>函数
	grunt.registerTask("async-task", function () {
		const done = this.async();
		setTimeout(() => {
			// ...
			done(); // 调用这个done才结束task
		}, 3000);
	});
};

task 失败

  • 在 task 中return false标记 task 失败 
    grup.registerTask("fail-task", () => {
	//...
	return false;
});
  • 在异步 task 中给 done 传 false 参数 
    grup.registerTask("fail-async-task", function () {
	const done = this.async();
	//...
	setTimeout(() => {
		//...
		done(false);
	}, 3000);
});
  • 如果是一个 task 数组，会在失败处中断
  • 可以加上--force强行执行所有 task

task 配置

module.exports = (grunt) => {
	grunt.initConfig({
		foo: {
			bar: "666",
		},
	});

	grunt.registerTask("foo", () => {
		// 获取配置
		console.log(grunt.config()); // { foo: { bar: '666' } }
		console.log(grunt.config("foo")); // { bar: '666' }
		console.log(grunt.config("foo.bar")); // 666
	});
};

多目标任务

grunt.initConfig({
	"multi-task1": {
		options: {
			// 任务配置选项
			foo: "bar",
		},
		// 除了options外都会注册为目标
		t1: "1",
		t2: {
			options: {
				foo: "baz",
			},
			name: "target222",
		},
	},
});
grunt.registerMultiTask("multi-task1", function () {
	console.log(this.options(), this.data.options ? "options被覆盖" : "无覆盖的options");
	console.log(`target=>${this.target};data=>${this.data.name}`);
	// Running "multi:t1" (multi) task
	// { foo: 'bar' } 无覆盖的options
	// target=>t1;data=>undefined

	// Running "multi:t2" (multi) task
	// { foo: 'baz' } options被覆盖
	// target=>t2;data=>target222
});
  • 只执行单个目标
yarn gulp multi-task1:[目标名]

load-grunt-tasks

This module will read the dependencies/devDependencies/peerDependencies/optionalDependencies in your package.json and load grunt tasks that match the provided patterns.

  • 使用
// 上面的grunt-sass可以改为
const load = require("load-grunt-tasks");
load(grunt); // 加载dependencies/devDependencies/peerDependencies/optionalDependencies中所有tasks
load();
  • 选项
require("load-grunt-tasks")(grunt, {
	pattern: "grunt-contrib-*", //  正则
	config: "../package.json",
	scope: "devDependencies", // load范围
	requireResolution: true, // 在全部文件中查找而不仅是 node_modules
});

Gulp

基于流的构建工具

默认任务都是异步的，需要手动调用结束任务

// gulpfile.js
exports.foo = (done) => {
	setTimeout(() => {
		done();
	}, 2000);
};

// 默认任务，yarn gulp就行
exports.default = (done) => {
	//...
	done();
};

// 标记失败
exports.failTask = (done) => {
	done(new Error("task fail"));
};

串行任务

按顺序执行，阻塞

const { series } = require("gulp");
exports.s1 = series("t1", "t2");

并行任务

同时执行

const { parallel } = require("gulp");
exports.p1 = series("t1", "t2");

异步操作

  • promise

    // 支持promise
exports.p1 = () => {
	return Promise.resolve();
};
exports.p1Fail = () => {
	return Promise.reject(new Error("task fail"));
};

  • async/await

    // 支持async await
const timeout = (wait) => new Promise((resolve) => setTimeout(resolve, wait));
exports.a1 = async () => {
	await timeout(3000);
};

  • stream

    exports.s1 = () => {
	const readStream = fs.createReadStream("src/source.txt");
	const writeStream = fs.createWriteStream("dist/dist.txt");
	readStream.pipe(writeStream);
	return readStream;
};

    构建过程

graph LR
A[读取流] --> B[转换流] --> c[写入流]
const {src,dest} = require('gulp')
const cleanCss = require('gulp-clean-css')

exports.sample = () => {
  src('src/style/*.css') // 读取
    .pipe(cleanCss())                    //转换

    .pipe(dest('dist/css')) //  写入
}
